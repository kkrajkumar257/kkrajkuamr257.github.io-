Features of the Website

The website includes the following key features:

  1. Starting Balance: You begin with $100 billion to spend (equivalent to a major tech billionaire's fortune)

  3. Interactive Item Grid: A diverse collection of purchasable items ranging from everyday objects to luxury goods, organized in a visually appealing grid

How to Use

  1. Browse through the available items displayed on the page

  4. Watch your remaining balance update in real-time at the top of the page
  5. Try to spend the entire $100 billion!

        // Create items and add them to the container
        function createItems() {
            const container = document.getElementById('items-container');
            
            items.forEach(item => {
                // Create item quantities object if it doesn't exist
                if (!item.quantity) {
                    item.quantity = 0;
                }
                
                // Create item element
                const itemElement = document.createElement('div');
                itemElement.className = 'item';

        // Buy an item
        function buyItem(itemName) {
            // Find the item
            const item = items.find(i => i.name === itemName);
            if (!item) return;
            
            // Check if we have enough money
            if (balance >= item.price) {
                // Update balance and quantities
                balance -= item.price;
                spent += item.price;
                item.quantity += 1;
                
                // Update display
                updateItemDisplay(item);
                updateDisplay();
            }
        }
        
        // Sell an item
        function sellItem(itemName) {
            // Find the item
            const item = items.find(i => i.name === itemName);
            if (!item) return;
            
            // Check if we have the item
            if (item.quantity > 0) {
                // Update balance and quantities
                balance += item.price;
                spent -= item.price;
                item.quantity -= 1;
                
                // Update display
                updateItemDisplay(item);
                updateDisplay();
            }
        }
        
        // Update the display of a specific item
        function updateItemDisplay(item) {
            // Find all items with the same name
            const itemElements = document.querySelectorAll('.item');
            
            itemElements.forEach(element => {
                const nameElement = element.querySelector('.item-name');
                
                if (nameElement && nameElement.textContent === item.name) {
                    // Update quantity
                    element.querySelector('.quantity').textContent = item.quantity;
                    
                    // Update button states
                    const sellButton = element.querySelector('.sell-btn');
                    sellButton.disabled = item.quantity === 0;
                    
                    const buyButton = element.querySelector('.buy-btn');
                    buyButton.disabled = balance < item.price;
                }
            });
        }

            items.forEach(item => updateItemDisplay(item));
        }
